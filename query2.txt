//director and genre

Select * FROM movies x, countries y, genres z
WHERE
x.gID = z.genreID
AND
x.producedIn = y.countryID

//director and release date

SELECT * FROM directorIDs x, directorMovies y, movies z
WHERE
x.directorID = y.directorMovieID
AND
y.movID = z.movieID
AND
z.releaseDate BETWEEN '"1999-05-05"' AND '"2001-03-07"'
